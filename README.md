`Slist* slist_reverselist(Slist *list){`
    `Node *prev = NULL;`
    `Node *next = NULL;`
    `list->tail = list->head;`

    `while(list->head != NULL) {`
        `next = list->head->next;`
        `list->head->next = prev;`
        `prev = list->head;`
        `list->head = next;`
    `}`
    `list->head = prev;`
    `return list;`
`}`
